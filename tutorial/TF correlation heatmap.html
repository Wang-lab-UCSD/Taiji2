# define function
get_tf_correlation <- function(sample,regulatee){
    df <- read.csv(paste0('edge_weight_per_locus_all_',sample,'.csv'))
    candidateTFs <- df |> filter(X.END_ID == regulatee) |> group_by(X.START_ID) |> summarise(n=n()) |> slice_max(order_by = n, n=30, with_ties = T) |> pull(X.START_ID) ## select the top 10 TFs with most binding sites
    
    get_tf_corr_helper <- function(tf1, tf2){
        df1 <- df |> filter(X.END_ID == regulatee & X.START_ID == tf1) |> select(chr, start.int, end.int)
        names(df1) <- c('chrom','start','end')
        df2 <- df |> filter(X.END_ID == regulatee & X.START_ID == tf2) |> select(chr, start.int, end.int)
        names(df2) <- c('chrom','start','end')
        df3 <- bed_intersect(df1, df2) |> filter(.overlap>0) |> distinct(chrom, start.x, end.x) 
        dim(df3)[1]/dim(df1)[1]
    }
    data <- mapply(get_tf_corr_helper, rep(candidateTFs, each = length(candidateTFs)), rep(candidateTFs, length(candidateTFs)))
    data2 <- data.frame('tf1' = rep(candidateTFs, each = length(candidateTFs)),
                    'tf2' = rep(candidateTFs, length(candidateTFs)),
                    'cov' = unlist(data)) |> tidyr::pivot_wider(names_from = 'tf2', values_from = 'cov') |> tibble::column_to_rownames('tf1')

    return(data2)
}

# define function v2: add TF of interest
get_tf_correlation_v2 <- function(sample,regulatee,add_tfs){
    df <- read.csv(paste0('edge_weight_per_locus_all_',sample,'.csv'))
    candidateTFs <- df |> filter(X.END_ID == regulatee) |> group_by(X.START_ID) |> summarise(n=n()) |> slice_max(order_by = n, n=30, with_ties = T) |> pull(X.START_ID) ## select the top 10 TFs with most binding sites
    
    # add TF of interest
    candidateTFs <- union(candidateTFs,add_tfs)
    
    get_tf_corr_helper <- function(tf1, tf2){
        df1 <- df |> filter(X.END_ID == regulatee & X.START_ID == tf1) |> select(chr, start.int, end.int)
        names(df1) <- c('chrom','start','end')
        df2 <- df |> filter(X.END_ID == regulatee & X.START_ID == tf2) |> select(chr, start.int, end.int)
        names(df2) <- c('chrom','start','end')
        df3 <- bed_intersect(df1, df2) |> filter(.overlap>0) |> distinct(chrom, start.x, end.x) 
        dim(df3)[1]/dim(df1)[1]
    }
    data <- mapply(get_tf_corr_helper, rep(candidateTFs, each = length(candidateTFs)), rep(candidateTFs, length(candidateTFs)))
    data2 <- data.frame('tf1' = rep(candidateTFs, each = length(candidateTFs)),
                    'tf2' = rep(candidateTFs, length(candidateTFs)),
                    'cov' = unlist(data)) |> tidyr::pivot_wider(names_from = 'tf2', values_from = 'cov') |> tibble::column_to_rownames('tf1')

    return(data2)
}
